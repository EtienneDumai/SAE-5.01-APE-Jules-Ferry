import { Component, inject } from '@angular/core';
import {ToastService} from '../../services/Toast/toast.service';
import { toSignal } from '@angular/core/rxjs-interop';

@Component({
  selector: 'app-toast',
  standalone: true,
  imports: [],
  templateUrl: './toast.component.html',
  styleUrl: './toast.component.css'
})
export class ToastComponent {
  private readonly toastService = inject(ToastService);
  toast = toSignal(this.toastService.toast, { initialValue: null });
  close(): void {
    this.toastService.clear();
  }
}

@if (toast(); as toast) {
<div
  class="flex flex-wrap justify-between w-32 fixed bottom-4 right-4 z-50 rounded w-fit max-w-sm items-center shadow-2xl transition-shadow"
  [class]="toast.type"
>
  @if (toast.type === 'error') {
  <div class="text-white rounded-lg px-4 py-3 flex items-center gap-3">
    <p class="text-sm leading-snug break-all">{{ toast.message }}</p>

}
